class Solution {
  public:
    // Function to find pair with maximum sum
    int pairWithMaxSum(vector<int> &arr) {
        // Your code goes here
        int i=0;
        int j=1;
        int n = arr.size();
        int temp=0;
        int maxSum =0;
        for(int m=0;m<n-1;m++){
            temp=arr[i++]+arr[j++];
            maxSum = max(maxSum,temp);
            temp=0;
        }
        
        return maxSum;
    }
};
