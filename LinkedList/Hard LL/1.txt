/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    int length(ListNode*head){
        ListNode*temp=head;
        int len=0;
        while(temp!=NULL)
        {
            len++;
            temp=temp->next;
        }
        return len;
    }
    ListNode* reverseKGroup(ListNode* head, int k) {
        if(head==NULL)
        {
            return head;
        }
        if(head->next==NULL)
        {
            return head;
        }
        ListNode*curr=head;
        ListNode*prev=NULL;
        ListNode*nextNode=curr->next;
        int pos=1;
        int len=length(head);
        if(len<k){
            return head;
        }
        while(pos<=k){
            nextNode=curr->next;
            curr->next=prev;
            prev=curr;
            curr=nextNode;
            pos++;
        }
        ListNode*recursionKans;
        if(nextNode!=NULL)
        {
            recursionKans= reverseKGroup(nextNode,k);
            head->next=recursionKans;
        }
        return prev;
    }
};
